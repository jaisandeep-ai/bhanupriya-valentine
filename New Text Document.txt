/* SLIDESHOW */
let slideIndex = 0;
function startSlideshow() {
    const slides = document.querySelectorAll(".slide");

    setInterval(() => {
        slides[slideIndex].classList.remove("active");
        slideIndex++;

        if(slideIndex >= slides.length) {
            slideIndex = slides.length - 1;
            slides[slideIndex].classList.add("finale");
            document.getElementById("proposal").style.display = "block";
        }

        slides[slideIndex].classList.add("active");

    }, 8000);
}

/* HEART FIREWORKS */
function startHeartFireworks() {
    const canvas=document.getElementById("fireworks");
    const ctx=canvas.getContext("2d");
    canvas.width=window.innerWidth;
    canvas.height=window.innerHeight;

    function drawHeart(x,y,size,color,alpha){
        ctx.globalAlpha=alpha;
        ctx.fillStyle=color;
        ctx.beginPath();
        ctx.moveTo(x,y);
        ctx.bezierCurveTo(x-size,y-size,x-size*2,y+size,x,y+size*2);
        ctx.bezierCurveTo(x+size*2,y+size,x+size,y-size,x,y);
        ctx.fill();
        ctx.globalAlpha=1;
    }

    let hearts=[];

    function explode(){
        let x=Math.random()*canvas.width;
        let y=Math.random()*canvas.height/2;
        for(let i=0;i<20;i++){
            hearts.push({
                x:x,
                y:y,
                dx:(Math.random()-0.5)*4,
                dy:(Math.random()-0.5)*4,
                size:Math.random()*8+5,
                alpha:1,
                color:"gold"
            });
        }
    }

    function animate(){
        ctx.clearRect(0,0,canvas.width,canvas.height);
        hearts.forEach((h,i)=>{
            h.x+=h.dx;
            h.y+=h.dy;
            h.alpha-=0.02;
            drawHeart(h.x,h.y,h.size,h.color,h.alpha);
            if(h.alpha<=0) hearts.splice(i,1);
        });
        requestAnimationFrame(animate);
    }

    setInterval(explode,2000);
    animate();
}
